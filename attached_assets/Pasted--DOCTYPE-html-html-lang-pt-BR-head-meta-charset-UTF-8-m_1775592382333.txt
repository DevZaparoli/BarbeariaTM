const ADMIN_PASS   = '1234';   // ← altere para sua senha
const ALL_SLOTS    = ['07:00','08:00','09:00','10:00','11:00','12:00',
                      '13:00','14:00','15:00','16:00','17:00','18:00','19:00','20:00'];
const DEFAULT_OPEN = ['08:00','09:00','10:00','11:00','14:00','15:00','16:00','17:00','18:00'];
const MONTHS       = ['Janeiro','Fevereiro','Março','Abril','Maio','Junho',
                      'Julho','Agosto','Setembro','Outubro','Novembro','Dezembro'];
const WDAYS        = ['Dom','Seg','Ter','Qua','Qui','Sex','Sáb'];

/* ════════════════════════════════
   BANCO DE HORÁRIOS (em memória)
   scheduleDB['YYYY-MM-DD'] = ['08:00','09:00',...]  (slots DISPONÍVEIS)
════════════════════════════════ */
const scheduleDB = {};

function dateKey(d){ return d.toISOString().split('T')[0]; }
function getSlots(d){ const k=dateKey(d); return scheduleDB[k]!==undefined ? scheduleDB[k] : [...DEFAULT_OPEN]; }
function setSlots(k,arr){ scheduleDB[k]=arr; }

/* ════════════════════════════════
   ESTADO CLIENTE
════════════════════════════════ */
const today = new Date();
const S = { services:[], date:null, time:null, vy:today.getFullYear(), vm:today.getMonth() };

/* ── Serviços ── */
document.querySelectorAll('.service-card').forEach(c=>{
  c.addEventListener('click',()=>{
    const n=c.dataset.name, p=parseInt(c.dataset.price);
    const i=S.services.findIndex(x=>x.name===n);
    if(i>-1){ S.services.splice(i,1); c.classList.remove('selected'); }
    else { S.services.push({name:n,price:p}); c.classList.add('selected'); }
    syncSteps(); syncSummary();
  });
});

/* ── Calendário ── */
function renderCal(){
  const {vy,vm}=S;
  document.getElementById('calTitle').textContent = MONTHS[vm];
  document.getElementById('calYear').textContent  = vy;
  document.getElementById('monthLabel').textContent = `${MONTHS[vm]} de ${vy}`;

  const firstDow = new Date(vy,vm,1).getDay();
  const daysInM  = new Date(vy,vm+1,0).getDate();
  const todayMid = new Date(today.getFullYear(),today.getMonth(),today.getDate());
  const grid = document.getElementById('calDays');
  grid.innerHTML='';

  // células vazias
  for(let i=0;i<firstDow;i++){
    const e=document.createElement('div'); e.className='cal-day cal-empty'; grid.appendChild(e);
  }

  for(let d=1;d<=daysInM;d++){
    const cd  = new Date(vy,vm,d);
    const past= cd < todayMid;
    const dow = cd.getDay();
    const isSun= dow===0;
    const isSat= dow===6;
    const isTod= d===today.getDate() && vm===today.getMonth() && vy===today.getFullYear();
    const isSel= S.date && S.date.getDate()===d && S.date.getMonth()===vm && S.date.getFullYear()===vy;

    const cell=document.createElement('div');
    cell.className='cal-day';
    if(past||isSun) cell.classList.add('cal-past');
    if(isTod)  cell.classList.add('cal-today');
    if(isSun)  cell.classList.add('cal-sun');
    if(isSat)  cell.classList.add('cal-sat');
    if(isSel)  cell.classList.add('cal-selected');

    // número
    const num=document.createElement('div'); num.className='cal-day-num'; num.textContent=d; cell.appendChild(num);
    // label dow
    const wl=document.createElement('div'); wl.className='cal-day-wday'; wl.textContent=WDAYS[dow]; cell.appendChild(wl);

    // dots disponibilidade
    if(!past && !isSun){
      const slots=getSlots(cd);
      const dotsWrap=document.createElement('div'); dotsWrap.className='cal-day-dots';
      if(slots.length===0){
        const dt=document.createElement('div'); dt.className='cal-dot cal-dot-busy'; dotsWrap.appendChild(dt);
      } else {
        const show=Math.min(slots.length,3);
        for(let k=0;k<show;k++){ const dt=document.createElement('div'); dt.className='cal-dot'; dotsWrap.appendChild(dt); }
      }
      cell.appendChild(dotsWrap);
    }

    if(!past && !isSun){
      cell.style.cursor='pointer';
      cell.addEventListener('click',()=>{
        S.date=new Date(vy,vm,d); S.time=null;
        renderCal(); renderTimes(); syncSummary(); syncSteps();
        document.getElementById('timesSection').style.display='block';
        document.getElementById('timesSection').scrollIntoView({behavior:'smooth',block:'nearest'});
      });
    }
    grid.appendChild(cell);
  }
}

document.getElementById('prevBtn').addEventListener('click',()=>{ S.vm--; if(S.vm<0){S.vm=11;S.vy--;} renderCal(); });
document.getElementById('nextBtn').addEventListener('click',()=>{ S.vm++; if(S.vm>11){S.vm=0;S.vy++;} renderCal(); });

/* ── Horários cliente ── */
function renderTimes(){
  if(!S.date) return;
  document.getElementById('selectedDateLabel').textContent =
    S.date.toLocaleDateString('pt-BR',{weekday:'long',day:'numeric',month:'long'});
  const avail = getSlots(S.date);
  const grid  = document.getElementById('timesGrid');
  grid.innerHTML='';

  if(avail.length===0){
    grid.innerHTML='<p style="grid-column:1/-1;font-size:0.82rem;color:var(--muted);letter-spacing:1px;padding:12px 0;">Nenhum horário disponível para este dia. Escolha outra data.</p>';
    return;
  }

  ALL_SLOTS.forEach(t=>{
    const ok=avail.includes(t), sel=S.time===t;
    const s=document.createElement('div');
    s.className='time-slot'+(ok?'':' busy')+(sel?' selected':'');
    s.textContent=t;
    if(ok){ s.addEventListener('click',()=>{ S.time=t; renderTimes(); syncSummary(); syncSteps();
      document.getElementById('clientSection').style.display='block';
      document.getElementById('clientSection').scrollIntoView({behavior:'smooth',block:'nearest'}); }); }
    grid.appendChild(s);
  });
}

/* ── Summary & Steps ── */
function syncSummary(){
  const names=S.services.map(x=>x.name).join(' + ');
  const total=S.services.reduce((a,x)=>a+x.price,0);
  document.getElementById('sumService').textContent=names||'—';
  document.getElementById('sumDate').textContent=S.date?S.date.toLocaleDateString('pt-BR',{weekday:'short',day:'numeric',month:'long',year:'numeric'}):'—';
  document.getElementById('sumTime').textContent=S.time||'—';
  document.getElementById('sumPrice').textContent=total>0?`R$ ${total},00`:'—';
}
function syncSteps(){
  const hs=S.services.length>0, hd=!!S.date, ht=!!S.time;
  document.getElementById('step1').className='step'+(hs?' done':' active');
  document.getElementById('step2').className='step'+(hd?' done':(hs?' active':''));
  document.getElementById('step3').className='step'+(ht?' done':(hd?' active':''));
  document.getElementById('step4').className='step'+(ht?' active':'');
}

function confirmBooking(){
  const name=document.getElementById('clientName').value.trim();
  const phone=document.getElementById('clientPhone').value.trim();
  if(!S.services.length){alert('Selecione ao menos um serviço.');return;}
  if(!S.date){alert('Selecione uma data.');return;}
  if(!S.time){alert('Selecione um horário.');return;}
  if(!name){alert('Informe seu nome.');return;}
  if(!phone){alert('Informe seu telefone.');return;}

  // Remove horário da disponibilidade
  const k=dateKey(S.date);
  setSlots(k, getSlots(S.date).filter(t=>t!==S.time));

  const names=S.services.map(x=>x.name).join(' + ');
  const total=S.services.reduce((a,x)=>a+x.price,0);
  const ds=S.date.toLocaleDateString('pt-BR',{weekday:'long',day:'numeric',month:'long',year:'numeric'});
  document.getElementById('successDetail').innerHTML=`<strong>${name}</strong><br>${names}<br>${ds} às ${S.time}<br><strong>R$ ${total},00</strong>`;
  document.getElementById('successOverlay').classList.add('show');
}

  document.getElementById('timesSection').style.display='none';
  document.getElementById('clientSection').style.display='none';
  renderCal(); syncSummary(); syncSteps();

}

function loadAdminDay(){
  const dp=document.getElementById('adminDatePick');
  if(!dp.value) return;
  const [y,m,d]=dp.value.split('-').map(Number);
  const dt=new Date(y,m-1,d);
  const badge=document.getElementById('adminDayBadge');
  badge.textContent=WDAYS[dt.getDay()]; badge.style.display='inline-block';

  // Inicializa estado temporário
  const cur=getSlots(dt);
  adminTmp={};
  ALL_SLOTS.forEach(s=>{ adminTmp[s]=cur.includes(s)?'avail':'blocked'; });

  renderAdminGrid();
  document.getElementById('adminDayWrap').style.display='block';
  document.getElementById('saveStatus').className='save-status';
}

function renderAdminGrid(){
  const grid=document.getElementById('adminTimesGrid');
  grid.innerHTML='';
  ALL_SLOTS.forEach(t=>{
    const st=adminTmp[t]||'blocked';
    const div=document.createElement('div');
    div.className=`adm-slot ${st}`;

    const span1=document.createElement('span'); span1.className='adm-slot-time'; span1.textContent=t;
    const span2=document.createElement('span'); span2.className='adm-slot-status';
    span2.textContent=st==='avail'?'✓ disponível':'✕ bloqueado';

    div.appendChild(span1); div.appendChild(span2);
    div.addEventListener('click',()=>{
      adminTmp[t]=adminTmp[t]==='avail'?'blocked':'avail';
      renderAdminGrid();
      document.getElementById('saveStatus').className='save-status';
    });
    grid.appendChild(div);
  });
}

function bulkSet(st){ ALL_SLOTS.forEach(s=>{ adminTmp[s]=st; }); renderAdminGrid(); document.getElementById('saveStatus').className='save-status'; }
function bulkBusiness(){
  ALL_SLOTS.forEach(s=>{ adminTmp[s]='blocked'; });
  DEFAULT_OPEN.forEach(s=>{ if(adminTmp.hasOwnProperty(s)) adminTmp[s]='avail'; });
  renderAdminGrid();
  document.getElementById('saveStatus').className='save-status';
}

function saveDay(){
  const dp=document.getElementById('adminDatePick');
  if(!dp.value) return;
  const avail=Object.entries(adminTmp).filter(([,v])=>v==='avail').map(([k])=>k);
  setSlots(dp.value,avail);
  renderCal();
  if(S.date && dateKey(S.date)===dp.value) renderTimes();
  document.getElementById('saveStatus').className='save-status ok';
}
